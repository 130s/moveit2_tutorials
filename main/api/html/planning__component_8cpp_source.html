115 planning_interface::MotionPlanResponse PlanningComponent::plan(const PlanRequestParameters& parameters,
116  const bool store_solution)
117 {
118  auto plan_solution = planning_interface::MotionPlanResponse();
119 
120  // check if joint_model_group exists
121  if (!joint_model_group_)
122  {
123  RCLCPP_ERROR(LOGGER, "Failed to retrieve joint model group for name '%s'.", group_name_.c_str());
124  plan_solution.error_code_ = moveit::core::MoveItErrorCode::INVALID_GROUP_NAME;
125  if (store_solution)
126  {
127  last_plan_solution_ = plan_solution;
128  }
129  return plan_solution;
130  }
131 
132  // Clone current planning scene
133  planning_scene_monitor::PlanningSceneMonitorPtr planning_scene_monitor = moveit_cpp_->getPlanningSceneMonitor();
134  planning_scene_monitor->updateFrameTransforms();
135  const planning_scene::PlanningScenePtr planning_scene = [planning_scene_monitor] {
136  planning_scene_monitor::LockedPlanningSceneRO ls(planning_scene_monitor);
137  return planning_scene::PlanningScene::clone(ls);
138  }();
139  planning_scene_monitor.reset(); // release this pointer
140 
141  // Init MotionPlanRequest
142  ::planning_interface::MotionPlanRequest req;
143  req.group_name = group_name_;
144  req.planner_id = parameters.planner_id;
145  req.num_planning_attempts = std::max(1, parameters.planning_attempts);
146  req.allowed_planning_time = parameters.planning_time;
147  req.max_velocity_scaling_factor = parameters.max_velocity_scaling_factor;
148  req.max_acceleration_scaling_factor = parameters.max_acceleration_scaling_factor;
149  if (workspace_parameters_set_)
150  req.workspace_parameters = workspace_parameters_;
151 
152  // Set start state
153  moveit::core::RobotStatePtr start_state = considered_start_state_;
154  if (!start_state)
155  start_state = moveit_cpp_->getCurrentState();
156  start_state->update();
157  moveit::core::robotStateToRobotStateMsg(*start_state, req.start_state);
158  planning_scene->setCurrentState(*start_state);
159 
160  // Set goal constraints
161  if (current_goal_constraints_.empty())
162  {
163  RCLCPP_ERROR(LOGGER, "No goal constraints set for planning request");
164  plan_solution.error_code_ = moveit::core::MoveItErrorCode::INVALID_GOAL_CONSTRAINTS;
165  if (store_solution)
166  {
167  last_plan_solution_ = plan_solution;
168  }
169  return plan_solution;
170  }
171  req.goal_constraints = current_goal_constraints_;
172  // Set path constraints
173  req.path_constraints = current_path_constraints_;
174  // Set trajectory constraints
175  req.trajectory_constraints = current_trajectory_constraints_;
176 
177  // Run planning attempt
178  ::planning_interface::MotionPlanResponse res;
179  const auto& pipelines = moveit_cpp_->getPlanningPipelines();
180  auto it = pipelines.find(parameters.planning_pipeline);
181  if (it == pipelines.end())
182  {
183  RCLCPP_ERROR(LOGGER, "No planning pipeline available for name '%s'", parameters.planning_pipeline.c_str());
184  plan_solution.error_code_ = moveit::core::MoveItErrorCode::FAILURE;
185  if (store_solution)
186  {
187  last_plan_solution_ = plan_solution;
188  }
189  return plan_solution;
190  }
191  const planning_pipeline::PlanningPipelinePtr pipeline = it->second;
192  pipeline->generatePlan(planning_scene, req, res);
193 
194  plan_solution.error_code_ = res.error_code_;
195  if (res.error_code_.val != res.error_code_.SUCCESS)
196  {
197  RCLCPP_ERROR(LOGGER, "Could not compute plan successfully");
198  if (store_solution)
199  {
200  last_plan_solution_ = plan_solution;
201  }
202  return plan_solution;
203  }
204  plan_solution.trajectory_ = res.trajectory_;
205  plan_solution.planning_time_ = res.planning_time_;
206  plan_solution.start_state_ = req.start_state;
207  plan_solution.error_code_ = res.error_code_.val;
208 
209  // TODO(henningkayser): Visualize trajectory
210  // std::vector<const moveit::core::LinkModel*> eef_links;
211  // if (joint_model_group->getEndEffectorTips(eef_links))
212  //{
213  // for (const auto& eef_link : eef_links)
214  // {
215  // RCLCPP_INFO_STREAM("Publishing trajectory for end effector " << eef_link->getName());
216  // visual_tools_->publishTrajectoryLine(last_solution_trajectory_, eef_link);
217  // visual_tools_->publishTrajectoryPath(last_solution_trajectory_, false);
218  // visual_tools_->publishRobotState(last_solution_trajectory_->getLastWayPoint(), rviz_visual_tools::TRANSLUCENT);
219  // }
220  //}
221 
222  if (store_solution)
223  {
224  last_plan_solution_ = plan_solution;
225  }
226  return plan_solution;
227 }

229 planning_interface::MotionPlanResponse
230 PlanningComponent::plan(const MultiPipelinePlanRequestParameters& parameters,
231  const SolutionCallbackFunction& solution_selection_callback,
232  StoppingCriterionFunction stopping_criterion_callback)
233 {
234  // Create solutions container
235  PlanSolutions planning_solutions{ parameters.multi_plan_request_parameters.size() };
236  std::vector<std::thread> planning_threads;
237  planning_threads.reserve(parameters.multi_plan_request_parameters.size());
238 
239  // Print a warning if more parallel planning problems than available concurrent threads are defined. If
240  // std::thread::hardware_concurrency() is not defined, the command returns 0 so the check does not work
241  auto const hardware_concurrency = std::thread::hardware_concurrency();
242  if (parameters.multi_plan_request_parameters.size() > hardware_concurrency && hardware_concurrency != 0)
243  {
244  RCLCPP_WARN(
245  LOGGER,
246  "More parallel planning problems defined ('%ld') than possible to solve concurrently with the hardware ('%d')",
247  parameters.multi_plan_request_parameters.size(), hardware_concurrency);
248  }
249 
250  // Launch planning threads
251  for (const auto& plan_request_parameter : parameters.multi_plan_request_parameters)
252  {
253  auto planning_thread = std::thread([&]() {
254  auto plan_solution = planning_interface::MotionPlanResponse();
255  try
256  {
257  plan_solution = plan(plan_request_parameter, false);
258  }
259  catch (const std::exception& e)
260  {
261  RCLCPP_ERROR_STREAM(LOGGER, "Planning pipeline '" << plan_request_parameter.planning_pipeline.c_str()
262  << "' threw exception '" << e.what() << "'");
263  plan_solution = planning_interface::MotionPlanResponse();
264  plan_solution.error_code_ = moveit::core::MoveItErrorCode::FAILURE;
265  }
266  plan_solution.planner_id_ = plan_request_parameter.planner_id;
267  planning_solutions.pushBack(plan_solution);
268 
269  if (stopping_criterion_callback != nullptr)
270  {
271  if (stopping_criterion_callback(planning_solutions, parameters))
272  {
273  // Terminate planning pipelines
274  RCLCPP_ERROR_STREAM(LOGGER, "Stopping criterion met: Terminating planning pipelines that are still active");
275  for (const auto& plan_request_parameter : parameters.multi_plan_request_parameters)
276  {
277  moveit_cpp_->terminatePlanningPipeline(plan_request_parameter.planning_pipeline);
278  }
279  }
280  }
281  });
282  planning_threads.push_back(std::move(planning_thread));
283  }
284 
285  // Wait for threads to finish
286  for (auto& planning_thread : planning_threads)
287  {
288  if (planning_thread.joinable())
289  {
290  planning_thread.join();
291  }
292  }
293 
294  // Return best solution determined by user defined callback (Default: Shortest path)
295  last_plan_solution_ = solution_selection_callback(planning_solutions.getSolutions());
296  return last_plan_solution_;
297 }
