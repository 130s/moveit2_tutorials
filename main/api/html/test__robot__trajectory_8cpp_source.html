486  initTestTrajectory(trajectory);
487 
488  ASSERT_EQ(5u, trajectory->size());
489  std::vector<double> positions;
490 
491  double start_pos = 0.0;
492 
493  for (size_t i = 0; i < trajectory->size(); ++i)
494  {
495  auto waypoint = trajectory->getWayPointPtr(i);
496  // modify joint values
497  waypoint->copyJointGroupPositions(arm_jmg_name_, positions);
498  start_pos = positions[0];
499  positions[0] += 0.01 * i;
500  waypoint->setJointGroupPositions(arm_jmg_name_, positions);
501  }
502 
503  unsigned int count = 0;
504  for (const auto& waypoint_and_duration : *trajectory)
505  {
506  const auto& waypoint = waypoint_and_duration.first;
507  waypoint->copyJointGroupPositions(arm_jmg_name_, positions);
508  EXPECT_EQ(start_pos + count * 0.01, positions[0]);
509  count++;
510  }
511 
512  EXPECT_EQ(count, trajectory->size());
513 
514  // Consistency checks
515  EXPECT_EQ(trajectory->begin(), trajectory->begin());
516  EXPECT_EQ(trajectory->end(), trajectory->end());
517 
518  // trajectory has length 5; incrementing begin 5 times should reach the end
519  EXPECT_NE(trajectory->begin(), trajectory->end());
520  EXPECT_NE(++trajectory->begin(), trajectory->end());
521  EXPECT_NE(++(++trajectory->begin()), trajectory->end());
